    05_еӯ—з¬ҰдёІж“ҚдҪңиҜҰи§Ј¶

    жқҘжәҗ: PythonйқўиҜ•е®һи·ө/00_йӣ¶еҹәзЎҖе…Ҙй—Ё/05_еӯ—з¬ҰдёІж“ҚдҪңиҜҰи§Ј.py

    `python

    ============================================================¶

    Python йӣ¶еҹәзЎҖе…Ҙй—Ё вҖ”вҖ” з¬¬5иҜҫпјҡеӯ—з¬ҰдёІж“ҚдҪңиҜҰи§Ј¶

    ============================================================¶

    ============================================================¶

    1. еӯ—з¬ҰдёІеҹәзЎҖ¶

    ============================================================¶

    print("=" * 50) print("з¬¬дёҖйғЁеҲҶпјҡеӯ—з¬ҰдёІеҹәзЎҖ") print("=" * 50)

    еӯ—з¬ҰдёІеҸҜд»Ҙз”ЁеҚ•еј•еҸ·жҲ–еҸҢеј•еҸ·еҲӣе»ә¶

    s1 = 'Hello' s2 = "World" print(f"s1 = {s1}, s2 = {s2}")

    еҰӮжһңеӯ—з¬ҰдёІеҶ…йғЁжңүеј•еҸ·пјҢз”ЁеҸҰдёҖз§Қеј•еҸ·еҢ…иЈ№¶

    sentence = "I'm a student" # еӨ–йғЁз”ЁеҸҢеј•еҸ·пјҢеҶ…йғЁеҚ•еј•еҸ·жІЎй—®йўҳ sentence2 = 'He said "Hi"' # еӨ–йғЁз”ЁеҚ•еј•еҸ·пјҢеҶ…йғЁеҸҢеј•еҸ·жІЎй—®йўҳ print(sentence) print(sentence2)

    еӨҡиЎҢеӯ—з¬ҰдёІз”Ёдёүеј•еҸ·¶

    multi_line = """ иҝҷжҳҜз¬¬дёҖиЎҢ иҝҷжҳҜз¬¬дәҢиЎҢ иҝҷжҳҜз¬¬дёүиЎҢ """ print(f"\nеӨҡиЎҢеӯ—з¬ҰдёІ:\n{multi_line}")

    ============================================================¶

    2. еӯ—з¬ҰдёІзҙўеј•е’ҢеҲҮзүҮ¶

    ============================================================¶

    print("\n" + "=" * 50) print("з¬¬дәҢйғЁеҲҶпјҡзҙўеј•е’ҢеҲҮзүҮ") print("=" * 50)

    s = "Python" print(f"еӯ—з¬ҰдёІ: {s}")

    зҙўеј•пјҡд»Һ 0 ејҖе§Ӣи®Ўж•°¶

    print(f"s[0] = {s[0]}") # P print(f"s[1] = {s[1]}") # y print(f"s[-1] = {s[-1]}") # nпјҲеҖ’ж•°з¬¬1дёӘпјү print(f"s[-2] = {s[-2]}") # oпјҲеҖ’ж•°з¬¬2дёӘпјү

    еҲҮзүҮпјҡ[ејҖе§Ӣ:з»“жқҹ:жӯҘй•ҝ] жіЁж„Ҹпјҡз»“жқҹдҪҚзҪ®дёҚеҢ…еҗ«¶

    print(f"\ns[0:3] = {s[0:3]}") # PytпјҲзҙўеј• 0,1,2пјү print(f"s[:3] = {s[:3]}") # PytпјҲд»ҺејҖеӨҙеҲ°2пјү print(f"s[2:] = {s[2:]}") # thonпјҲд»Һ2еҲ°жң«е°ҫпјү print(f"s[1:5] = {s[1:5]}") # ythoпјҲзҙўеј• 1,2,3,4пјү print(f"s[::2] = {s[::2]}") # PtoпјҲжҜҸйҡ”дёҖдёӘеҸ–пјү print(f"s[::-1] = {s[::-1]}") # nohtyPпјҲеҸҚиҪ¬пјҒйқўиҜ•еёёиҖғпјү

    еӯ—з¬ҰдёІжҳҜдёҚеҸҜеҸҳзҡ„пјҒ¶

    s[0] = "J" # жҠҘй”ҷпјҒдёҚиғҪдҝ®ж”№еӯ—з¬ҰдёІ¶

    ============================================================¶

    3. еёёз”Ёеӯ—з¬ҰдёІж–№жі•¶

    ============================================================¶

    print("\n" + "=" * 50) print("з¬¬дёүйғЁеҲҶпјҡеёёз”Ёеӯ—з¬ҰдёІж–№жі•") print("=" * 50)

    text = " Hello, Python World! " print(f"еҺҹеӯ—з¬ҰдёІ: '{text}'")

    --- еӨ§е°ҸеҶҷиҪ¬жҚў ---¶

    print(f"\ntext.upper() = '{text.upper()}'") # е…ЁеӨ§еҶҷ print(f"text.lower() = '{text.lower()}'") # е…Ёе°ҸеҶҷ print(f"text.title() = '{text.title()}'") # жҜҸдёӘеҚ•иҜҚйҰ–еӯ—жҜҚеӨ§еҶҷ

    --- еҺ»йҷӨз©әзҷҪ ---¶

    print(f"text.strip() = '{text.strip()}'") # еҺ»жҺүдёӨиҫ№з©әж ј print(f"text.lstrip() = '{text.lstrip()}'") # еҺ»жҺүе·Ұиҫ№з©әж ј print(f"text.rstrip() = '{text.rstrip()}'") # еҺ»жҺүеҸіиҫ№з©әж ј

    --- жҹҘжүҫе’ҢжӣҝжҚў ---¶

    print(f"\ntext.find('Python') = {text.find('Python')}") # 7пјҲйҰ–ж¬ЎеҮәзҺ°зҡ„дҪҚзҪ®пјү print(f"text.find('Java') = {text.find('Java')}") # -1пјҲжІЎжүҫеҲ°пјү print(f"text.replace('Python', 'Java') = '{text.replace('Python', 'Java')}'")

    --- еҲӨж–ӯзұ»ж–№жі• ---¶

    print(f"\n'text'.isdigit() = {'123'.isdigit()}") # TrueпјҲе…ЁжҳҜж•°еӯ—пјү print(f"'abc'.isalpha() = {'abc'.isalpha()}") # TrueпјҲе…ЁжҳҜеӯ—жҜҚпјү print(f"'abc123'.isalnum() = {'abc123'.isalnum()}") # TrueпјҲеӯ—жҜҚжҲ–ж•°еӯ—пјү print(f"'hello'.startswith('he') = {'hello'.startswith('he')}") # True print(f"'hello'.endswith('lo') = {'hello'.endswith('lo')}") # True

    --- еҲҶеүІе’ҢиҝһжҺҘ ---¶

    print(f"\n'a,b,c'.split(',') = {'a,b,c'.split(',')}") # ['a', 'b', 'c'] print(f"'hello world'.split() = {'hello world'.split()}") # ['hello', 'world']

    words = ["Python", "is", "great"] print(f"' '.join(words) = {' '.join(words)}") # "Python is great" print(f"'-'.join(words) = {'-'.join(words)}") # "Python-is-great"

    --- з»ҹи®Ў ---¶

    print(f"\n'hello'.count('l') = {'hello'.count('l')}") # 2 print(f"len('hello') = {len('hello')}") # 5

    ============================================================¶

    4. еӯ—з¬ҰдёІж јејҸеҢ–пјҲйқўиҜ•еҝ…дјҡпјү¶

    ============================================================¶

    print("\n" + "=" * 50) print("з¬¬еӣӣйғЁеҲҶпјҡеӯ—з¬ҰдёІж јејҸеҢ–пјҲйқўиҜ•еҝ…дјҡпјү") print("=" * 50)

    name = "еј дёү" age = 20 score = 85.5

    ж–№жі•1пјҡf-stringпјҲжңҖжҺЁиҚҗпјү¶

    print(f"\nж–№жі•1пјҡf-string") print(f"жҲ‘еҸ«{name}пјҢд»Ҡе№ҙ{age}еІҒпјҢиҖғдәҶ{score}еҲҶ") print(f"еҲҶж•°дҝқз•ҷ1дҪҚе°Ҹж•°пјҡ{score:.1f}") # 85.5 print(f"ж•ҙж•°иЎҘйӣ¶пјҡ{age:05d}") # 00020 print(f"еҸіеҜ№йҪҗе®ҪеәҰ10пјҡ{name:>10}") # " еј дёү" print(f"е·ҰеҜ№йҪҗе®ҪеәҰ10пјҡ{name:<10}") # "еј дёү " print(f"еұ…дёӯеҜ№йҪҗе®ҪеәҰ10пјҡ{name:^10}") # " еј дёү "

    ж–№жі•2пјҡformat()¶

    print(f"\nж–№жі•2пјҡformat()") print("жҲ‘еҸ«{}пјҢд»Ҡе№ҙ{}еІҒ".format(name, age)) print("жҲ‘еҸ«{0}пјҢд»Ҡе№ҙ{1}еІҒпјҢ{0}жҳҜеӯҰз”ҹ".format(name, age)) # еҸҜд»ҘеӨҚз”Ё print("жҲ‘еҸ«{n}пјҢд»Ҡе№ҙ{a}еІҒ".format(n=name, a=age)) # е‘ҪеҗҚеҸӮж•°

    ж–№жі•3пјҡ% ж јејҸеҢ–пјҲйқўиҜ•еҸҜиғҪйҒҮеҲ°ж—§д»Јз Ғпјү¶

    print(f"\nж–№жі•3пјҡ% ж јејҸеҢ–") print("жҲ‘еҸ«%sпјҢд»Ҡе№ҙ%dеІҒпјҢиҖғдәҶ%.1fеҲҶ" % (name, age, score))

    %s еӯ—з¬ҰдёІпјҢ%d ж•ҙж•°пјҢ%f жө®зӮ№ж•°пјҢ%.1f дҝқз•ҷ1дҪҚе°Ҹж•°¶

    print("жҚўиЎҢпјҡ\nиҝҷжҳҜз¬¬дәҢиЎҢ") print("еҲ¶иЎЁз¬ҰпјҡеҲ—1\tеҲ—2\tеҲ—3") print("еҸҚж–ңжқ жң¬иә«пјҡ\") print("еҸҢеј•еҸ·пјҡ\"Hello\"")
