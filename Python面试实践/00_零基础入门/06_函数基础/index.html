    06_函数基础¶

    来源: Python面试实践/00_零基础入门/06_函数基础.py

    # ============================================================
# Python 零基础入门 —— 第6课：函数基础
# ============================================================


# ============================================================
# 1. 什么是函数？
# ============================================================

print("=" * 50)
print("第一部分：什么是函数？")
print("=" * 50)

"""
函数 = 一段可以重复使用的代码块
类比：就像洗衣机的一个程序
  - "标准洗"程序：进水 -> 洗涤 -> 漂洗 -> 脱水
  - 每次洗衣服，只需要按一下"标准洗"按钮，不用每次都手动操作

在 Python 中：
  - 先"定义"函数（编写程序）
  - 然后"调用"函数（按下按钮）
"""

# --- 定义函数 ---
# def 函数名():
#     函数体（要执行的代码）

def say_hello():
    """这是一个打招呼的函数"""
    print("你好！")
    print("很高兴见到你！")

# --- 调用函数 ---
print("\n--- 调用 say_hello() ---")
say_hello()
say_hello()  # 可以重复调用！


# ============================================================
# 2. 带参数的函数
# ============================================================

print("\n" + "=" * 50)
print("第二部分：带参数的函数")
print("=" * 50)

# 参数 = 函数需要的外部信息
# 类比：洗衣机程序需要知道"水温多少度"

def greet(name):
    """向指定的人打招呼"""
    print(f"你好，{name}！")

print("\n--- 带一个参数 ---")
greet("张三")
greet("李四")

# 多个参数用逗号分隔
def introduce(name, age):
    """自我介绍"""
    print(f"我叫{name}，今年{age}岁")

print("\n--- 带两个参数 ---")
introduce("王五", 25)
introduce("赵六", 30)


# ============================================================
# 3. 返回值 return
# ============================================================

print("\n" + "=" * 50)
print("第三部分：返回值 return")
print("=" * 50)

# 返回值 = 函数执行完后，把结果"交出来"
# 类比：洗衣机洗完衣服，把洗好的衣服"还给你"

def add(a, b):
    """计算两个数的和"""
    result = a + b
    return result  # 把结果返回给调用者

print("\n--- 使用返回值 ---")
sum1 = add(3, 5)
print(f"3 + 5 = {sum1}")

sum2 = add(10, 20)
print(f"10 + 20 = {sum2}")

# 返回值可以直接使用，不用先存到变量
print(f"7 + 8 = {add(7, 8)}")

# 没有 return 的函数，默认返回 None
def say_hi():
    print("Hi")

result = say_hi()
print(f"没有 return 的函数返回: {result}")  # None


# ============================================================
# 4. 默认参数
# ============================================================

print("\n" + "=" * 50)
print("第四部分：默认参数")
print("=" * 50)

# 默认参数 = 调用时如果不传，就用默认值
# 类比：洗衣机默认水温是 30度，你不改就按 30度洗

def greet_with_title(name, title="同学"):
    """打招呼，带称呼，默认是 同学"""
    print(f"{title} {name}，你好！")

print("\n--- 使用默认参数 ---")
greet_with_title("张三")             # 不传 title，用默认值"同学"
greet_with_title("李四", "老师")     # 传了 title，用传入的值

# 【面试陷阱】默认参数如果是可变对象，会被共享！
print("\n--- 默认参数陷阱 ---")

def add_item(item, items=[]):
    items.append(item)
    return items

print(f"第一次调用: {add_item(1)}")   # [1]
print(f"第二次调用: {add_item(2)}")   # [1, 2]  ← 坑！共享了同一个列表！

# 正确写法：用 None 作为默认值
def add_item_safe(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

print(f"\n安全写法第一次: {add_item_safe(1)}")   # [1]
print(f"安全写法第二次: {add_item_safe(2)}")   # [2]  ← 正确！


# ============================================================
# 5. 关键字参数
# ============================================================

print("\n" + "=" * 50)
print("第五部分：关键字参数")
print("=" * 50)

# 调用函数时，可以指定参数名，不用管顺序

def create_user(name, age, city):
    print(f"姓名:{name}, 年龄:{age}, 城市:{city}")

print("\n--- 位置调用（按顺序）---")
create_user("张三", 20, "北京")

print("\n--- 关键字调用（按名字）---")
create_user(age=20, city="北京", name="张三")  # 顺序可以打乱！

# 混合使用：位置参数在前，关键字参数在后
create_user("李四", city="上海", age=25)


# ============================================================
# 6. 可变参数 *args 和 **kwargs
# ============================================================

print("\n" + "=" * 50)
print("第六部分：可变参数")
print("=" * 50)

# --- *args：接收任意多个位置参数 ---
print("\n--- *args 接收任意多个数 ---")

def sum_all(*numbers):
    """计算任意多个数的和"""
    total = 0
    for n in numbers:
        total += n
    return total

print(f"sum_all(1, 2, 3) = {sum_all(1, 2, 3)}")
print(f"sum_all(10, 20) = {sum_all(10, 20)}")
print(f"sum_all() = {sum_all()}")  # 不传也可以

# --- **kwargs：接收任意多个关键字参数 ---
print("\n--- **kwargs 接收任意多个关键字参数 ---")

def print_info(**kwargs):
    """打印任意多个键值对"""
    for key, value in kwargs.items():
        print(f"  {key}: {value}")

print_info(name="张三", age=20)
print("---")
print_info(name="李四", age=25, city="上海", hobby="编程")


# ============================================================
# 7. 函数文档注释
# ============================================================

print("\n" + "=" * 50)
print("第七部分：函数文档")
print("=" * 50)

def calculate_area(length, width):
    """
    计算矩形的面积

    参数:
        length (float): 长
        width (float): 宽

    返回:
        float: 面积
    """
    return length * width

# 查看函数文档
print(calculate_area.__doc__)
print(f"\n面积 = {calculate_area(5, 3)}")


# ============================================================
# 8. 局部变量和全局变量
# ============================================================

print("\n" + "=" * 50)
print("第八部分：作用域")
print("=" * 50)

# 局部变量：函数内部定义的变量，函数外访问不到
def demo_local():
    local_var = 100
    print(f"函数内部: {local_var}")

demo_local()
# print(local_var)  # 报错！函数外部访问不到

# 全局变量：函数外部定义的变量
global_var = 200

def demo_global():
    print(f"函数内部读取全局变量: {global_var}")

demo_global()

# 如果要在函数内部修改全局变量，需要用 global 声明
def modify_global():
    global global_var  # 声明我要用的是全局变量
    global_var = 300
    print(f"修改后: {global_var}")

modify_global()
print(f"函数外部查看: {global_var}")  # 300，确实被修改了

# 【建议】尽量少用 global，容易让代码难以理解和调试


# ============================================================
# 9. 面试考点
# ============================================================

print("\n" + "=" * 50)
print("面试考点")
print("=" * 50)

print("""
Q: 什么是函数？为什么用函数？
A: 函数是封装了一段可重复使用的代码块。
   好处：减少重复代码、便于维护、逻辑清晰。

Q: return 和 print 的区别？
A: print 是把内容显示到屏幕上，不影响程序执行
   return 是把值交还给调用者，调用者可以拿到这个值继续用

Q: *args 和 **kwargs 的区别？
A: *args 接收任意多个位置参数，内部是元组
   **kwargs 接收任意多个关键字参数，内部是字典

Q: 默认参数陷阱是什么？
A: 默认参数如果是可变对象（如 list、dict），
   会在函数定义时只创建一次，所有调用共享同一个对象！
   解决：默认用 None，函数内部判断后创建新对象。

Q: 局部变量和全局变量的区别？
A: 局部变量只在函数内部有效，函数结束就销毁
   全局变量在整个文件有效，但函数内修改需要 global 声明
""")
