    07_еҠЁжҖҒи§„еҲ’_зҲ¬жҘјжўҜ¶

    жқҘжәҗ: PythonйқўиҜ•е®һи·ө/03_йқўиҜ•й«ҳйў‘жүӢж’•йўҳ/07_еҠЁжҖҒи§„еҲ’_зҲ¬жҘјжўҜ.py

    """
гҖҗйқўиҜ•й«ҳйў‘гҖ‘еҠЁжҖҒи§„еҲ’ вҖ”вҖ” зҲ¬жҘјжўҜй—®йўҳ

йўҳзӣ®пјҡеҒҮи®ҫдҪ жӯЈеңЁзҲ¬жҘјжўҜгҖӮйңҖиҰҒ n йҳ¶дҪ жүҚиғҪеҲ°иҫҫжҘјйЎ¶гҖӮ
      жҜҸж¬ЎдҪ еҸҜд»ҘзҲ¬ 1 жҲ– 2 дёӘеҸ°йҳ¶гҖӮй—®жңүеӨҡе°‘з§ҚдёҚеҗҢзҡ„ж–№жі•еҸҜд»ҘзҲ¬еҲ°жҘјйЎ¶пјҹ

зӨәдҫӢпјҡ
  n = 2 -> 2 з§ҚпјҲ1+1, 2пјү
  n = 3 -> 3 з§ҚпјҲ1+1+1, 1+2, 2+1пјү
  n = 4 -> 5 з§Қ

ж ёеҝғжҖқжғіпјҲеҠЁжҖҒи§„еҲ’пјүпјҡ
  зҲ¬еҲ°з¬¬ n йҳ¶зҡ„ж–№жі• = зҲ¬еҲ°з¬¬ n-1 йҳ¶зҡ„ж–№жі•пјҲеҶҚиҝҲ1жӯҘпјү
                     + зҲ¬еҲ°з¬¬ n-2 йҳ¶зҡ„ж–№жі•пјҲеҶҚиҝҲ2жӯҘпјү
  еҚіпјҡf(n) = f(n-1) + f(n-2)
  иҝҷе°ұжҳҜж–җжіўйӮЈеҘ‘ж•°еҲ—пјҒ
"""


# ============================================================
# и§Јжі•1пјҡйҖ’еҪ’ + и®°еҝҶеҢ–пјҲиҮӘйЎ¶еҗ‘дёӢпјү
# ============================================================

def climb_stairs_memo(n: int) -> int:
    """
    и®°еҝҶеҢ–йҖ’еҪ’ вҖ”вҖ” йқўиҜ•жҺЁиҚҗзҡ„з¬¬дёҖз§ҚеҶҷжі•

    жҖқи·Ҝпјҡ
    1. жҠҠеӨ§й—®йўҳжӢҶжҲҗе°Ҹй—®йўҳпјҲзҲ¬еҲ° n йҳ¶ = зҲ¬еҲ° n-1 + зҲ¬еҲ° n-2пјү
    2. з”Ёзј“еӯҳи®°дҪҸе·Із»Ҹз®—иҝҮзҡ„з»“жһңпјҢйҒҝе…ҚйҮҚеӨҚи®Ўз®—
    3. йҖ’еҪ’з»Ҳжӯўпјҡn=1 жңү 1 з§Қж–№жі•пјҢn=2 жңү 2 з§Қж–№жі•
    """
    # memo еӯ—е…ёпјҡи®°еҪ•е·Із»Ҹи®Ўз®—иҝҮзҡ„еҖј
    # key = йҳ¶ж•°пјҢvalue = ж–№жі•ж•°
    memo = {}

    def dp(i):
        # йҖ’еҪ’з»ҲжӯўжқЎд»¶пјҲжңҖеҹәжң¬зҡ„жғ…еҶөпјү
        if i <= 2:
            return i  # i=1 иҝ”еӣһ 1пјҢi=2 иҝ”еӣһ 2

        # еҰӮжһңе·Із»Ҹи®Ўз®—иҝҮпјҢзӣҙжҺҘиҝ”еӣһзј“еӯҳз»“жһң
        if i in memo:
            return memo[i]

        # зҠ¶жҖҒиҪ¬з§»ж–№зЁӢпјҡf(i) = f(i-1) + f(i-2)
        result = dp(i - 1) + dp(i - 2)

        # жҠҠз»“жһңеӯҳе…Ҙзј“еӯҳ
        memo[i] = result
        return result

    return dp(n)


# ============================================================
# и§Јжі•2пјҡеҠЁжҖҒи§„еҲ’ вҖ”вҖ” иҮӘеә•еҗ‘дёҠеЎ«иЎЁпјҲйқўиҜ•жңҖжҺЁиҚҗпјҒпјү
# ============================================================

def climb_stairs_dp(n: int) -> int:
    """
    еҠЁжҖҒи§„еҲ’ вҖ”вҖ” иҮӘеә•еҗ‘дёҠ

    жҖқи·Ҝпјҡ
    1. е®ҡд№ү dp[i] = зҲ¬еҲ°з¬¬ i йҳ¶зҡ„ж–№жі•ж•°
    2. еҲқе§ӢеҢ–пјҡdp[1] = 1, dp[2] = 2
    3. йҖ’жҺЁпјҡdp[i] = dp[i-1] + dp[i-2]
    4. д»Һ 3 ејҖе§ӢдёҖзӣҙз®—еҲ° n
    """
    if n <= 2:
        return n

    # dp ж•°з»„пјҢdp[i] иЎЁзӨәзҲ¬еҲ°з¬¬ i йҳ¶зҡ„ж–№жі•ж•°
    # зҙўеј• 0 дёҚз”ЁпјҢд»Һ 1 ејҖе§Ӣ
    dp = [0] * (n + 1)
    dp[1] = 1  # зҲ¬еҲ°з¬¬1йҳ¶еҸӘжңү1з§Қж–№жі•пјҡиҝҲ1жӯҘ
    dp[2] = 2  # зҲ¬еҲ°з¬¬2йҳ¶жңү2з§Қж–№жі•пјҡ1+1 жҲ– 2

    # д»Һз¬¬3йҳ¶ејҖе§ӢпјҢйҖҗдёӘи®Ўз®—
    for i in range(3, n + 1):
        # зҠ¶жҖҒиҪ¬з§»ж–№зЁӢ
        dp[i] = dp[i - 1] + dp[i - 2]

    return dp[n]


# ============================================================
# и§Јжі•3пјҡз©әй—ҙдјҳеҢ– вҖ”вҖ” еҸӘз”ЁдёӨдёӘеҸҳйҮҸпјҲйқўиҜ•еҠ еҲҶйЎ№пјү
# ============================================================

def climb_stairs_optimized(n: int) -> int:
    """
    з©әй—ҙдјҳеҢ–зүҲ вҖ”вҖ” O(1) з©әй—ҙеӨҚжқӮеәҰ

    жҖқи·Ҝпјҡ
    и®Ўз®— dp[i] еҸӘйңҖиҰҒ dp[i-1] е’Ң dp[i-2]пјҢдёҚйңҖиҰҒдҝқеӯҳжӣҙж—©зҡ„еҖјгҖӮ
    жүҖд»ҘеҸӘйңҖиҰҒдёӨдёӘеҸҳйҮҸж»ҡеҠЁжӣҙж–°еҚіеҸҜгҖӮ
    """
    if n <= 2:
        return n

    # prev2 иЎЁзӨә dp[i-2]пјҢprev1 иЎЁзӨә dp[i-1]
    prev2, prev1 = 1, 2

    for i in range(3, n + 1):
        # еҪ“еүҚеҖј = еүҚдёӨдёӘеҖјд№Ӣе’Ң
        current = prev1 + prev2

        # ж»ҡеҠЁжӣҙж–°пјҡprev2 еҸҳжҲҗ prev1пјҢprev1 еҸҳжҲҗ current
        prev2 = prev1
        prev1 = current

    return prev1


# ============================================================
# еҸҳдҪ“пјҡжҜҸж¬ЎеҸҜд»Ҙиө° 1гҖҒ2 жҲ– 3 жӯҘ
# ============================================================

def climb_stairs_three_steps(n: int) -> int:
    """
    еҸҳдҪ“пјҡжҜҸж¬ЎеҸҜд»ҘиҝҲ 1гҖҒ2 жҲ– 3 жӯҘ

    зҠ¶жҖҒиҪ¬з§»ж–№зЁӢеҸҳжҲҗпјҡ
    f(n) = f(n-1) + f(n-2) + f(n-3)
    """
    if n == 1:
        return 1
    if n == 2:
        return 2  # 1+1, 2
    if n == 3:
        return 4  # 1+1+1, 1+2, 2+1, 3

    dp = [0] * (n + 1)
    dp[1], dp[2], dp[3] = 1, 2, 4

    for i in range(4, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3]

    return dp[n]

"""
гҖҗйқўиҜ•иҝҪй—®гҖ‘

Q: д»Җд№ҲжҳҜеҠЁжҖҒи§„еҲ’пјҲDPпјүпјҹ
A: еҠЁжҖҒи§„еҲ’ = жҠҠеӨ§й—®йўҳжӢҶжҲҗе°Ҹй—®йўҳ + и®°дҪҸе°Ҹй—®йўҳзҡ„зӯ”жЎҲйҒҝе…ҚйҮҚеӨҚи®Ўз®—гҖӮ
   ж ёеҝғдёүиҰҒзҙ пјҡзҠ¶жҖҒе®ҡд№үгҖҒзҠ¶жҖҒиҪ¬з§»ж–№зЁӢгҖҒеҲқе§ӢжқЎд»¶гҖӮ

Q: зҲ¬жҘјжўҜдёәд»Җд№ҲиғҪз”Ё DPпјҹ
A: еӣ дёәзҲ¬еҲ°з¬¬ n йҳ¶зҡ„ж–№жі•еҸӘе’Ңз¬¬ n-1гҖҒn-2 йҳ¶жңүе…іпјҢ
   жІЎжңүеҗҺж•ҲжҖ§пјҲжңӘжқҘзҡ„йҖүжӢ©дёҚеҪұе“ҚиҝҮеҺ»зҡ„зҠ¶жҖҒпјүгҖӮ

Q: иҮӘйЎ¶еҗ‘дёӢпјҲйҖ’еҪ’+и®°еҝҶеҢ–пјүе’ҢиҮӘеә•еҗ‘дёҠпјҲеЎ«иЎЁпјүзҡ„еҢәеҲ«пјҹ
A: - иҮӘйЎ¶еҗ‘дёӢпјҡд»ҺеӨ§й—®йўҳйҖ’еҪ’еҲ°е°Ҹй—®йўҳпјҢжӣҙзӣҙи§ӮпјҢдҪҶйҖ’еҪ’ж ҲжңүејҖй”Җ
   - иҮӘеә•еҗ‘дёҠпјҡд»ҺеҹәзЎҖжғ…еҶөйҖҗжӯҘжҺЁеҜјеҲ°еӨ§й—®йўҳпјҢж— йҖ’еҪ’ејҖй”ҖпјҢжӣҙй«ҳж•Ҳ
   йқўиҜ•ж—¶дјҳе…ҲеҶҷиҮӘеә•еҗ‘дёҠзүҲжң¬гҖӮ

Q: з©әй—ҙдјҳеҢ–жҖҺд№ҲеҒҡзҡ„пјҹ
A: и§ӮеҜҹзҠ¶жҖҒиҪ¬з§»ж–№зЁӢпјҢеҸ‘зҺ°и®Ўз®— dp[i] еҸӘйңҖиҰҒеүҚдёӨдёӘеҖјпјҢ
   дёҚйңҖиҰҒж•ҙдёӘж•°з»„гҖӮз”ЁдёӨдёӘеҸҳйҮҸж»ҡеҠЁжӣҙж–°еҚіеҸҜгҖӮ
