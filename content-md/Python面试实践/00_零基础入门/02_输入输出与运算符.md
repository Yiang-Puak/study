`python
# ============================================================
# Python 零基础入门 —— 第2课：输入输出与运算符
# ============================================================


# ============================================================
# 1. 输出：print() 函数
# ============================================================

print("=" * 50)
print("第一部分：输出 print()")
print("=" * 50)

# --- 1.1 基本输出 ---
print("\n--- 基本输出 ---")
print("你好，世界！")        # 直接输出字符串
print(42)                   # 输出数字
print(3.14)                 # 输出小数

# --- 1.2 同时输出多个值 ---
print("\n--- 同时输出多个值 ---")
print("姓名", "张三", "年龄", 20)  # 用逗号分隔，会自动加空格

# --- 1.3 格式化输出（面试必会） ---
print("\n--- 格式化输出 ---")

# 方法1：f-string（最推荐，Python 3.6+）
name = "张三"
age = 20
print(f"我叫{name}，今年{age}岁")  # f"...{变量}..."

# 方法2：format()
print("我叫{}，今年{}岁".format(name, age))

# 方法3：% 格式化（较老，但面试可能遇到）
print("我叫%s，今年%d岁" % (name, age))
# %s 表示字符串，%d 表示整数，%f 表示浮点数

# --- 1.4 控制小数位数 ---
pi = 3.1415926
print(f"\npi = {pi}")
print(f"pi 保留2位 = {pi:.2f}")     # 3.14
print(f"pi 保留4位 = {pi:.4f}")     # 3.1416

# --- 1.5 控制输出结尾 ---
print("\n--- 控制输出结尾 ---")
print("A", end="")      # end="" 表示不换行
print("B", end="")
print("C")              # 默认 end="\n"（换行）
# 输出结果：ABC

print("=" * 30, end="\n\n")  # 输出两个换行


# ============================================================
# 2. 输入：input() 函数
# ============================================================

print("=" * 50)
print("第二部分：输入 input()")
print("=" * 50)

# input() 函数：让用户从键盘输入内容
# 注意：input() 返回的永远是字符串！

print("\n--- 基本输入 ---")
# 取消注释下面两行可以运行：
# user_input = input("请输入你的名字：")
# print(f"你输入了：{user_input}")

print("""
【重要】input() 返回的永远是字符串！
  如果你输入 20，得到的是 "20"（字符串），不是 20（数字）
  需要自己做类型转换
""")

# 正确获取数字输入的方式：
# age_str = input("请输入年龄：")   # 得到字符串 "20"
# age = int(age_str)                # 转成整数 20
# 或者简写：
# age = int(input("请输入年龄："))

# 演示（不用真正输入）：
# 模拟用户输入了 "25"
simulated_input = "25"
age = int(simulated_input)
print(f"输入的字符串：'{simulated_input}' -> 转成整数：{age}")
print(f"明年你 {age + 1} 岁")


# ============================================================
# 3. 运算符
# ============================================================

print("\n" + "=" * 50)
print("第三部分：运算符")
print("=" * 50)

# --- 3.1 算术运算符 ---
print("\n--- 算术运算符 ---")
a, b = 17, 5
print(f"a = {a}, b = {b}")
print(f"a + b  = {a + b}")      # 加法：22
print(f"a - b  = {a - b}")      # 减法：12
print(f"a * b  = {a * b}")      # 乘法：85
print(f"a / b  = {a / b}")      # 除法：3.4（结果永远是浮点数）
print(f"a // b = {a // b}")     # 整除：3（只取整数部分，向下取整）
print(f"a % b  = {a % b}")      # 取余：2（17 除以 5 余 2）
print(f"a ** b = {a ** b}")     # 幂运算：17^5 = 1419857

print("""
【面试常问】/ 和 // 的区别：
  /  是普通除法，结果带小数：17 / 5 = 3.4
  // 是整除，只留整数部分：17 // 5 = 3（向下取整）
  %  是取余：17 % 5 = 2
""")

# 负数整除（面试陷阱）
print(f"-17 // 5 = {-17 // 5}")   # -4（向下取整，不是 -3！）

# --- 3.2 比较运算符 ---
print("\n--- 比较运算符 ---")
print(f"a == b: {a == b}")    # 等于
print(f"a != b: {a != b}")    # 不等于
print(f"a > b:  {a > b}")     # 大于
print(f"a < b:  {a < b}")     # 小于
print(f"a >= b: {a >= b}")    # 大于等于
print(f"a <= b: {a <= b}")    # 小于等于

# --- 3.3 逻辑运算符 ---
print("\n--- 逻辑运算符 ---")
# and：并且（两边都为 True，结果才为 True）
# or：或者（只要一边为 True，结果就为 True）
# not：非（取反）

x, y = True, False
print(f"True and False = {x and y}")   # False
print(f"True or  False = {x or y}")      # True
print(f"not True = {not x}")              # False

# 短路求值（面试考点）
print("""
【短路求值】
  a and b：如果 a 为 False，直接返回 a，不计算 b
  a or b：如果 a 为 True，直接返回 a，不计算 b
""")

# 实际例子
score = 85
print(f"分数 {score} 是否及格且优秀：{score >= 60 and score >= 90}")
# score >= 60 为 True，继续判断 score >= 90 为 False
# 最终结果为 False

# --- 3.4 赋值运算符 ---
print("\n--- 赋值运算符 ---")
c = 10
print(f"c = {c}")

c += 3   # 等价于 c = c + 3
print(f"c += 3 -> {c}")   # 13

c -= 5   # 等价于 c = c - 5
print(f"c -= 5 -> {c}")   # 8

c *= 2   # 等价于 c = c * 2
print(f"c *= 2 -> {c}")   # 16

c //= 3  # 等价于 c = c // 3
print(f"c //= 3 -> {c}")  # 5


# ============================================================
# 4. 运算符优先级
# ============================================================

print("\n--- 运算符优先级（从高到低）---")
print("""
  1. ()        括号最优先
  2. **        幂运算
  3. * / // %  乘除整除取余
  4. + -       加减
  5. == != > < >= <=  比较
  6. not       非
  7. and       并且
  8. or        或者
""")

# 例子
result = 2 + 3 * 4       # 先算 3*4=12，再 2+12=14
print(f"2 + 3 * 4 = {result}")  # 14，不是 20！

result2 = (2 + 3) * 4    # 括号改变优先级
print(f"(2 + 3) * 4 = {result2}")  # 20


# ============================================================
# 5. 练习题
# ============================================================

print("\n" + "=" * 50)
print("练习题")
print("=" * 50)

print("""
练习1：计算圆的面积
  radius = 5
  area = ?
  提示：圆的面积公式 = 3.14159 * radius * radius

练习2：温度转换
  摄氏温度 c = 25
  转成华氏温度：f = c * 9/5 + 32
  打印结果

练习3：判断闰年
  年份 year = 2024
  闰年条件：能被4整除但不能被100整除，或者能被400整除
  打印是否是闰年
""")

print("\n本节完毕！建议：把运算符部分自己算一遍，对比输出结果")
