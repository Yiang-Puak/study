`python
"""
【面试高频】动态规划 —— 爬楼梯问题

题目：假设你正在爬楼梯。需要 n 阶你才能到达楼顶。
      每次你可以爬 1 或 2 个台阶。问有多少种不同的方法可以爬到楼顶？

示例：
  n = 2 -> 2 种（1+1, 2）
  n = 3 -> 3 种（1+1+1, 1+2, 2+1）
  n = 4 -> 5 种

核心思想（动态规划）：
  爬到第 n 阶的方法 = 爬到第 n-1 阶的方法（再迈1步）
                     + 爬到第 n-2 阶的方法（再迈2步）
  即：f(n) = f(n-1) + f(n-2)
  这就是斐波那契数列！
"""


# ============================================================
# 解法1：递归 + 记忆化（自顶向下）
# ============================================================

def climb_stairs_memo(n: int) -> int:
    """
    记忆化递归 —— 面试推荐的第一种写法

    思路：
    1. 把大问题拆成小问题（爬到 n 阶 = 爬到 n-1 + 爬到 n-2）
    2. 用缓存记住已经算过的结果，避免重复计算
    3. 递归终止：n=1 有 1 种方法，n=2 有 2 种方法
    """
    # memo 字典：记录已经计算过的值
    # key = 阶数，value = 方法数
    memo = {}

    def dp(i):
        # 递归终止条件（最基本的情况）
        if i <= 2:
            return i  # i=1 返回 1，i=2 返回 2

        # 如果已经计算过，直接返回缓存结果
        if i in memo:
            return memo[i]

        # 状态转移方程：f(i) = f(i-1) + f(i-2)
        result = dp(i - 1) + dp(i - 2)

        # 把结果存入缓存
        memo[i] = result
        return result

    return dp(n)


# ============================================================
# 解法2：动态规划 —— 自底向上填表（面试最推荐！）
# ============================================================

def climb_stairs_dp(n: int) -> int:
    """
    动态规划 —— 自底向上

    思路：
    1. 定义 dp[i] = 爬到第 i 阶的方法数
    2. 初始化：dp[1] = 1, dp[2] = 2
    3. 递推：dp[i] = dp[i-1] + dp[i-2]
    4. 从 3 开始一直算到 n
    """
    if n <= 2:
        return n

    # dp 数组，dp[i] 表示爬到第 i 阶的方法数
    # 索引 0 不用，从 1 开始
    dp = [0] * (n + 1)
    dp[1] = 1  # 爬到第1阶只有1种方法：迈1步
    dp[2] = 2  # 爬到第2阶有2种方法：1+1 或 2

    # 从第3阶开始，逐个计算
    for i in range(3, n + 1):
        # 状态转移方程
        dp[i] = dp[i - 1] + dp[i - 2]

    return dp[n]


# ============================================================
# 解法3：空间优化 —— 只用两个变量（面试加分项）
# ============================================================

def climb_stairs_optimized(n: int) -> int:
    """
    空间优化版 —— O(1) 空间复杂度

    思路：
    计算 dp[i] 只需要 dp[i-1] 和 dp[i-2]，不需要保存更早的值。
    所以只需要两个变量滚动更新即可。
    """
    if n <= 2:
        return n

    # prev2 表示 dp[i-2]，prev1 表示 dp[i-1]
    prev2, prev1 = 1, 2

    for i in range(3, n + 1):
        # 当前值 = 前两个值之和
        current = prev1 + prev2

        # 滚动更新：prev2 变成 prev1，prev1 变成 current
        prev2 = prev1
        prev1 = current

    return prev1


# ============================================================
# 变体：每次可以走 1、2 或 3 步
# ============================================================

def climb_stairs_three_steps(n: int) -> int:
    """
    变体：每次可以迈 1、2 或 3 步

    状态转移方程变成：
    f(n) = f(n-1) + f(n-2) + f(n-3)
    """
    if n == 1:
        return 1
    if n == 2:
        return 2  # 1+1, 2
    if n == 3:
        return 4  # 1+1+1, 1+2, 2+1, 3

    dp = [0] * (n + 1)
    dp[1], dp[2], dp[3] = 1, 2, 4

    for i in range(4, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3]

    return dp[n]


# ============================================================
# 测试
# ========================

if __name__ == "__main__":
    print("=" * 50)
    print("爬楼梯 —— 动态规划入门")
    print("=" * 50)

    test_cases = [
        (1, 1),
        (2, 2),
        (3, 3),
        (4, 5),
        (5, 8),
        (10, 89),
    ]

    print("\n--- 标准爬楼梯（1或2步）---")
    for n, expected in test_cases:
        r1 = climb_stairs_memo(n)
        r2 = climb_stairs_dp(n)
        r3 = climb_stairs_optimized(n)
        status = "✅" if r1 == r2 == r3 == expected else "❌"
        print(f"{status} n={n:2d} | 记忆化: {r1:2d}, DP: {r2:2d}, 优化: {r3:2d} | 期望: {expected:2d}")

    print("\n--- 变体（1、2或3步）---")
    for n in range(1, 6):
        result = climb_stairs_three_steps(n)
        print(f"n={n} -> {result} 种方法")

"""
【面试追问】

Q: 什么是动态规划（DP）？
A: 动态规划 = 把大问题拆成小问题 + 记住小问题的答案避免重复计算。
   核心三要素：状态定义、状态转移方程、初始条件。

Q: 爬楼梯为什么能用 DP？
A: 因为爬到第 n 阶的方法只和第 n-1、n-2 阶有关，
   没有后效性（未来的选择不影响过去的状态）。

Q: 自顶向下（递归+记忆化）和自底向上（填表）的区别？
A: - 自顶向下：从大问题递归到小问题，更直观，但递归栈有开销
   - 自底向上：从基础情况逐步推导到大问题，无递归开销，更高效
   面试时优先写自底向上版本。

Q: 空间优化怎么做的？
A: 观察状态转移方程，发现计算 dp[i] 只需要前两个值，
   不需要整个数组。用两个变量滚动更新即可。

Q: 如果每次可以走 1~k 步呢？
A: f(n) = f(n-1) + f(n-2) + ... + f(n-k)
   用 k 个变量滚动更新，或者用一个长度为 k 的队列。

Q: DP 和递归+记忆化有什么区别？
A: 本质上是一样的，都是"拆分问题 + 缓存结果"。
   DP 是自底向上填表，递归记忆化是自顶向下求解。
"""
