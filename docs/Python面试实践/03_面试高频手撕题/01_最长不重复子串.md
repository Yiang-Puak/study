"""
【面试高频】最长不重复子串 —— 滑动窗口

题目：给定字符串 s，找出其中不含有重复字符的最长子串的长度。

示例：
  输入: "abcabcbb" → 输出: 3 ("abc")
  输入: "bbbbb" → 输出: 1 ("b")
  输入: "pwwkew" → 输出: 3 ("wke")

时间复杂度要求：O(n)
空间复杂度要求：O(min(m, n))，m 是字符集大小
"""


def length_of_longest_substring(s: str) -> int:
    """
    滑动窗口解法 —— 面试最推荐写法

    思路：
    1. 用左右指针维护一个窗口，窗口内字符不重复
    2. 右指针向右扩展，遇到重复字符时左指针收缩
    3. 用哈希表记录字符最新位置，实现 O(1) 查重

    面试要点：
    - 为什么用滑动窗口？—— 每个字符最多被访问两次（右指针一次，左指针一次），所以 O(n)
    - 为什么是 "最长子串" 不是 "子序列"？—— 子串要求连续，子序列不要求
    """
    char_index = {}  # 字符 -> 最新索引
    left = 0
    max_len = 0

    for right, char in enumerate(s):
        # 如果字符在窗口内重复了，收缩左边界
        if char in char_index and char_index[char] >= left:
            left = char_index[char] + 1

        # 更新字符位置
        char_index[char] = right

        # 更新最大长度
        max_len = max(max_len, right - left + 1)

    return max_len


def length_of_longest_substring_set(s: str) -> int:
    """
    滑动窗口 + Set 解法（另一种常见写法）

    思路：
    1. 用 set 维护窗口内字符
    2. 右指针扩展，遇到重复则从左边删除直到不重复
    """
    char_set = set()
    left = 0
    max_len = 0

    for right in range(len(s)):
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1
        char_set.add(s[right])
        max_len = max(max_len, right - left + 1)

    return max_len
