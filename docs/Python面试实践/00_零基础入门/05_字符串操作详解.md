```python
# ============================================================
# Python 零基础入门 —— 第5课：字符串操作详解
# ============================================================


# ============================================================
# 1. 字符串基础
# ============================================================

print("=" * 50)
print("第一部分：字符串基础")
print("=" * 50)

# 字符串可以用单引号或双引号创建
s1 = 'Hello'
s2 = "World"
print(f"s1 = {s1}, s2 = {s2}")

# 如果字符串内部有引号，用另一种引号包裹
sentence = "I'm a student"   # 外部用双引号，内部单引号没问题
sentence2 = 'He said "Hi"'    # 外部用单引号，内部双引号没问题
print(sentence)
print(sentence2)

# 多行字符串用三引号
multi_line = """
这是第一行
这是第二行
这是第三行
"""
print(f"\n多行字符串:\n{multi_line}")


# ============================================================
# 2. 字符串索引和切片
# ============================================================

print("\n" + "=" * 50)
print("第二部分：索引和切片")
print("=" * 50)

s = "Python"
print(f"字符串: {s}")

# 索引：从 0 开始计数
print(f"s[0] = {s[0]}")     # P
print(f"s[1] = {s[1]}")     # y
print(f"s[-1] = {s[-1]}")   # n（倒数第1个）
print(f"s[-2] = {s[-2]}")   # o（倒数第2个）

# 切片：[开始:结束:步长]  注意：结束位置不包含
print(f"\ns[0:3] = {s[0:3]}")    # Pyt（索引 0,1,2）
print(f"s[:3] = {s[:3]}")        # Pyt（从开头到2）
print(f"s[2:] = {s[2:]}")        # thon（从2到末尾）
print(f"s[1:5] = {s[1:5]}")      # ytho（索引 1,2,3,4）
print(f"s[::2] = {s[::2]}")      # Pto（每隔一个取）
print(f"s[::-1] = {s[::-1]}")    # nohtyP（反转！面试常考）

# 字符串是不可变的！
# s[0] = "J"  # 报错！不能修改字符串


# ============================================================
# 3. 常用字符串方法
# ============================================================

print("\n" + "=" * 50)
print("第三部分：常用字符串方法")
print("=" * 50)

text = "  Hello, Python World!  "
print(f"原字符串: '{text}'")

# --- 大小写转换 ---
print(f"\ntext.upper() = '{text.upper()}'")       # 全大写
print(f"text.lower() = '{text.lower()}'")       # 全小写
print(f"text.title() = '{text.title()}'")       # 每个单词首字母大写

# --- 去除空白 ---
print(f"text.strip() = '{text.strip()}'")         # 去掉两边空格
print(f"text.lstrip() = '{text.lstrip()}'")      # 去掉左边空格
print(f"text.rstrip() = '{text.rstrip()}'")      # 去掉右边空格

# --- 查找和替换 ---
print(f"\ntext.find('Python') = {text.find('Python')}")    # 7（首次出现的位置）
print(f"text.find('Java') = {text.find('Java')}")          # -1（没找到）
print(f"text.replace('Python', 'Java') = '{text.replace('Python', 'Java')}'")

# --- 判断类方法 ---
print(f"\n'text'.isdigit() = {'123'.isdigit()}")       # True（全是数字）
print(f"'abc'.isalpha() = {'abc'.isalpha()}")           # True（全是字母）
print(f"'abc123'.isalnum() = {'abc123'.isalnum()}")   # True（字母或数字）
print(f"'hello'.startswith('he') = {'hello'.startswith('he')}")  # True
print(f"'hello'.endswith('lo') = {'hello'.endswith('lo')}")      # True

# --- 分割和连接 ---
print(f"\n'a,b,c'.split(',') = {'a,b,c'.split(',')}")        # ['a', 'b', 'c']
print(f"'hello world'.split() = {'hello world'.split()}")      # ['hello', 'world']

words = ["Python", "is", "great"]
print(f"' '.join(words) = {' '.join(words)}")               # "Python is great"
print(f"'-'.join(words) = {'-'.join(words)}")               # "Python-is-great"

# --- 统计 ---
print(f"\n'hello'.count('l') = {'hello'.count('l')}")      # 2
print(f"len('hello') = {len('hello')}")                    # 5


# ============================================================
# 4. 字符串格式化（面试必会）
# ============================================================

print("\n" + "=" * 50)
print("第四部分：字符串格式化（面试必会）")
print("=" * 50)

name = "张三"
age = 20
score = 85.5

# 方法1：f-string（最推荐）
print(f"\n方法1：f-string")
print(f"我叫{name}，今年{age}岁，考了{score}分")
print(f"分数保留1位小数：{score:.1f}")       # 85.5
print(f"整数补零：{age:05d}")                # 00020
print(f"右对齐宽度10：{name:>10}")           # "        张三"
print(f"左对齐宽度10：{name:<10}")           # "张三        "
print(f"居中对齐宽度10：{name:^10}")         # "    张三    "

# 方法2：format()
print(f"\n方法2：format()")
print("我叫{}，今年{}岁".format(name, age))
print("我叫{0}，今年{1}岁，{0}是学生".format(name, age))  # 可以复用
print("我叫{n}，今年{a}岁".format(n=name, a=age))        # 命名参数

# 方法3：% 格式化（面试可能遇到旧代码）
print(f"\n方法3：% 格式化")
print("我叫%s，今年%d岁，考了%.1f分" % (name, age, score))
# %s 字符串，%d 整数，%f 浮点数，%.1f 保留1位小数


# ============================================================
# 5. 转义字符
# ============================================================

print("\n" + "=" * 50)
print("第五部分：转义字符")
print("=" * 50)

print("换行：\n这是第二行")
print("制表符：列1\t列2\t列3")
print("反斜杠本身：\\")
print("双引号：\"Hello\"")

# 原始字符串：r"..." 不处理转义
path = r"C:\Users\Documents\file.txt"
print(f"\n原始字符串: {path}")


# ============================================================
# 6. 面试考点
# ============================================================

print("\n" + "=" * 50)
print("面试考点")
print("=" * 50)

print("""
Q: 字符串是可变还是不可变的？
A: 不可变！任何"修改"字符串的操作都会返回新字符串，原字符串不变。

Q: 怎么判断一个字符串是不是回文？
A: s == s[::-1]

Q: 怎么去除字符串中的所有空格？
A: s.replace(" ", "")  或  "".join(s.split())

Q: 怎么把字符串中的单词顺序反转？
A: " ".join(s.split()[::-1])
   例："hello world" -> ["hello", "world"] -> ["world", "hello"] -> "world hello"

Q: f-string 和 format() 哪个更好？
A: f-string 更快、更易读，是 Python 3.6+ 的首选。
""")

print("\n" + "=" * 50)

print("=" * 50)

print("""

  统计每个单词出现的次数

  张三      20      85.5

  s = "a1b2c3"
  结果应为 "abc"

  s = "123"
  不调用 int()，自己写循环转换
```
