# 03_条件判断

> 来源: Python面试实践/00_零基础入门/03_条件判断.py

```python
# ============================================================
# Python 零基础入门 —— 第3课：条件判断
# ============================================================


# ============================================================
# 1. if 语句 —— 如果...就...
# ============================================================

print("=" * 50)
print("第一部分：if 语句")
print("=" * 50)

# 语法结构：
# if 条件:
#     条件为 True 时执行的代码
# 注意："if" 后面要加冒号，下面的代码要缩进！

print("\n--- 基本 if ---")
age = 20
if age >= 18:
    print("你已经成年了")
    print("可以考驾照了")
# 缩进的代码属于 if 块，条件满足才执行
# 不缩进的代码不属于 if 块，总是执行
print("这行不管条件如何都会执行")


# ============================================================
# 2. if...else —— 如果...就...否则...
# ============================================================

print("\n" + "=" * 50)
print("第二部分：if...else")
print("=" * 50)

# 语法结构：
# if 条件:
#     条件为 True 时执行
# else:
#     条件为 False 时执行

print("\n--- if...else ---")
score = 75
if score >= 60:
    print("及格了！")
else:
    print("不及格，需要补考")


# ============================================================
# 3. if...elif...else —— 多条件判断
# ============================================================

print("\n" + "=" * 50)
print("第三部分：if...elif...else")
print("=" * 50)

# 语法结构：
# if 条件1:
#     条件1满足时执行
# elif 条件2:
#     条件1不满足，但条件2满足时执行
# elif 条件3:
#     条件1、2不满足，但条件3满足时执行
# else:
#     以上条件都不满足时执行

print("\n--- 成绩等级判断 ---")
score = 82

if score >= 90:
    grade = "A"
elif score >= 80:
    grade = "B"
elif score >= 70:
    grade = "C"
elif score >= 60:
    grade = "D"
else:
    grade = "F"

print(f"分数 {score} -> 等级 {grade}")

# 【重要】elif 的顺序很重要！
# 如果写成 score >= 60 在前，score >= 90 在后，
# 那么 >=90 的分支永远走不到（因为 >=60 先满足就退出了）


# ============================================================
# 4. 条件判断的嵌套
# ============================================================

print("\n" + "=" * 50)
print("第四部分：嵌套条件")
print("=" * 50)

# 在 if 里面再写 if

print("\n--- 嵌套示例 ---")
is_member = True
purchase_amount = 200

if is_member:
    if purchase_amount >= 100:
        print("会员且满100，打8折")
    else:
        print("会员但未满100，打9折")
else:
    if purchase_amount >= 100:
        print("非会员但满100，打9折")
    else:
        print("非会员且未满100，无折扣")


# ============================================================
# 5. 简写：一行 if（三元运算符）
# ============================================================

print("\n" + "=" * 50)
print("第五部分：一行 if（三元运算符）")
print("=" * 50)

# 语法：值1 if 条件 else 值2
# 条件为 True 时取值1，否则取值2

age = 20
status = "成年" if age >= 18 else "未成年"
print(f"年龄 {age} -> {status}")

# 等价于：
# if age >= 18:
#     status = "成年"
# else:
#     status = "未成年"

# 嵌套三元运算符（面试可能遇到，但别写太多层）
score = 75
result = "优秀" if score >= 90 else "良好" if score >= 80 else "及格" if score >= 60 else "不及格"
print(f"分数 {score} -> {result}")


# ============================================================
# 6. 判断的真假规则
# ============================================================

print("\n" + "=" * 50)
print("第六部分：哪些值被认为是 False？")
print("=" * 50)

# 以下值在条件判断中被视为 False（"假值"）：
#   False, 0, 0.0, ""（空字符串）, []（空列表）, {}（空字典）, None

print("\n--- 假值测试 ---")
values = [False, 0, 0.0, "", [], {}, None]
for v in values:
    if v:
        print(f"{repr(v)} -> True")
    else:
        print(f"{repr(v)} -> False")

# 实用技巧：判断字符串是否为空
name = ""
if name:         # 等价于 if name != ""
    print(f"名字是 {name}")
else:
    print("名字为空")

# 判断列表是否为空
items = []
if items:        # 等价于 if len(items) > 0
    print("列表有内容")
else:
    print("列表为空")


# ============================================================
# 7. 逻辑运算符组合条件
# ============================================================

print("\n" + "=" * 50)
print("第七部分：组合条件")
print("=" * 50)

print("\n--- and（并且）---")
# 两边都要满足
temperature = 25
humidity = 60
if temperature >= 20 and temperature <= 30 and humidity < 70:
    print("天气舒适")

# 可以简写链式比较
if 20 <= temperature <= 30 and humidity < 70:
    print("天气舒适（链式写法）")

print("\n--- or（或者）---")
# 满足其一即可
day = "周六"
if day == "周六" or day == "周日":
    print("今天是周末")

# in 运算符（更简洁）
if day in ("周六", "周日"):
    print("今天是周末（in 写法）")

print("\n--- not（非）---")
is_closed = False
if not is_closed:
    print("门是开着的")


# ============================================================
# 8. is 与 == 的区别（面试高频！）
# ============================================================

print("\n" + "=" * 50)
print("第八部分：is 与 == 的区别")
print("=" * 50)

# == 比较"值"是否相等
# is 比较"内存地址"是否相同（是否是同一个对象）

a = [1, 2, 3]
b = [1, 2, 3]
c = a

print(f"a = {a}, b = {b}, c = a")
print(f"a == b: {a == b}")    # True（值相同）
print(f"a is b: {a is b}")    # False（不同对象，内存地址不同）
print(f"a is c: {a is c}")    # True（c 就是 a，同一个对象）

# 小整数和字符串会被缓存，所以 is 可能为 True
x = 100
y = 100
print(f"\n小整数：x == y: {x == y}, x is y: {x is y}")
# 注意：大整数不一定！

# 面试建议：判断值用 ==，判断身份用 is
# 常用：if x is None（判断是否为 None）


# ============================================================
# 9. 练习题
# ============================================================

print("\n" + "=" * 50)
print("练习题")
print("=" * 50)

print("""
练习1：判断奇偶
  num = 7
  判断是奇数还是偶数，打印结果
  提示：num % 2 == 0 表示偶数

练习2：判断三角形
  a, b, c = 3, 4, 5
  判断是否能构成三角形（两边之和大于第三边）
  如果能，再判断是直角、锐角还是钝角三角形

练习3：登录验证
  username = "admin"
  password = "123456"
  判断用户名和密码是否正确，给出不同提示

练习4：体重指数 BMI
  weight = 70  # kg
  height = 1.75  # m
  bmi = weight / (height ** 2)
  判断：BMI < 18.5 偏瘦，18.5-24 正常，24-28 偏胖，>=28 肥胖
""")

print("\n本节完毕！建议：把每个 if 例子的条件改成 False，看看输出变化")
```
