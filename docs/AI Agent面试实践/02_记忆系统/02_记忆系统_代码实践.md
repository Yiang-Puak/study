"""
【面试加分项】给 ReAct Agent 加上记忆系统 —— 代码实践

学习目标：
1. 实现短期记忆（对话历史维护 + 滑动窗口）
2. 实现摘要压缩（Buffer + Summary 混合策略）
3. 实现长期记忆（关键信息提取 + 向量化存储）
"""

import json
import os
from collections import deque
from typing import List, Dict, Optional
from openai import OpenAI


# ========================
# 第一步：短期记忆 —— 滑动窗口
# ========================

class ShortTermMemory:
    """
    短期记忆：维护最近 N 轮对话
    面试要点：为什么用 deque？—— 固定长度自动淘汰旧元素，O(1) 时间复杂度
    """

    def __init__(self, max_turns: int = 6):
        self.max_turns = max_turns
        self.messages = deque(maxlen=max_turns)

    def add(self, role: str, content: str, **kwargs):
        """添加一条消息"""
        msg = {"role": role, "content": content}
        msg.update(kwargs)
        self.messages.append(msg)

    def get_messages(self) -> List[Dict]:
        """获取当前记忆的所有消息"""
        return list(self.messages)

    def clear(self):
        """清空记忆"""
        self.messages.clear()


# ========================
# 第二步：摘要压缩记忆 —— Buffer + Summary
# ========================

class SummaryMemory:
    """
    摘要记忆：近期保留原始消息，早期生成摘要

    面试要点：
    - buffer_size：保留最近多少轮原始消息
    - 早期消息超过阈值后，调用 LLM 生成摘要
    - 摘要 + 近期原始消息 一起传给 LLM
    """

    def __init__(self, client: OpenAI, model: str, buffer_size: int = 4, summary_trigger: int = 8):
        self.client = client
        self.model = model
        self.buffer_size = buffer_size
        self.summary_trigger = summary_trigger  # 超过此轮数触发摘要
        self.raw_messages = []
        self.summary = ""

    def add(self, role: str, content: str):
        self.raw_messages.append({"role": role, "content": content})

        # 触发摘要压缩
        if len(self.raw_messages) > self.summary_trigger:
            self._compress()

    def _compress(self):
        """生成摘要，保留近期 buffer_size 条"""
        # 需要被摘要的部分
        to_summarize = self.raw_messages[:-self.buffer_size]
        recent = self.raw_messages[-self.buffer_size:]

        # 调用 LLM 生成摘要（实际项目中可用更便宜的模型）
        prompt = f"请用一段话概括以下对话的核心内容（保留关键事实和用户需求）：\n{json.dumps(to_summarize, ensure_ascii=False)}"
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}]
        )
        new_summary = response.choices[0].message.content

        # 合并旧摘要和新摘要
        if self.summary:
            self.summary = f"历史摘要：{self.summary}\n新摘要：{new_summary}"
        else:
            self.summary = new_summary

        self.raw_messages = recent
        print(f"[记忆压缩] 生成摘要：{self.summary[:80]}...")

    def get_messages(self) -> List[Dict]:
        """返回：系统摘要 + 近期原始消息"""
        result = []
        if self.summary:
            result.append({"role": "system", "content": f"【历史对话摘要】{self.summary}"})
        result.extend(self.raw_messages)
        return result


# ========================
# 第三步：长期记忆 —— 关键信息提取 + 向量存储
# ========================

class LongTermMemory:
    """
    长期记忆：提取关键信息，结构化存储

    面试要点：
    - 为什么不用纯向量存储？—— 关键信息（如用户偏好）需要精确召回，结构化更可靠
    - 向量存储适合"语义相似"检索，结构化存储适合"精确匹配"
    """

    def __init__(self, client: OpenAI, model: str):
        self.client = client
        self.model = model
        self.facts = {}  # 结构化事实：key = 实体/主题, value = 信息
        self.preferences = {}  # 用户偏好

    def extract_and_store(self, messages: List[Dict]):
        """从对话中提取关键信息并存储"""
        conversation = json.dumps(messages, ensure_ascii=False)
        prompt = f"""从以下对话中提取关键信息（用户偏好、重要事实），返回 JSON 格式：
{{
  "facts": {{"主题": "关键信息"}},
  "preferences": {{"偏好项": "具体值"}}
}}

对话内容：
{conversation}
"""
        try:
            response = self.client.chat.completions.create(
                model=self.model,
                messages=[{"role": "user", "content": prompt}],
                response_format={"type": "json_object"}
            )
            extracted = json.loads(response.choices[0].message.content)

            # 合并到已有记忆中
            self.facts.update(extracted.get("facts", {}))
            self.preferences.update(extracted.get("preferences", {}))
            print(f"[长期记忆] 提取到 facts: {extracted.get('facts', {})}")
        except Exception as e:
            print(f"[长期记忆] 提取失败: {e}")

    def get_context(self) -> str:
        """将长期记忆格式化为文本，注入系统提示"""
        parts = []
        if self.facts:
            parts.append("已知事实：" + json.dumps(self.facts, ensure_ascii=False))
        if self.preferences:
            parts.append("用户偏好：" + json.dumps(self.preferences, ensure_ascii=False))
        return "\n".join(parts) if parts else ""


# ========================
# 第四步：带记忆的 Agent
# ========================

class AgentWithMemory:
    """
    带三层记忆的 Agent：
    - 短期记忆：滑动窗口（当前会话最近 N 轮）
    - 中期记忆：摘要压缩（早期对话 → 摘要）
    - 长期记忆：关键信息提取（跨会话持久化）
    """

    def __init__(self, model: str = "qwen3-max"):
        self.client = OpenAI(
            api_key=os.getenv("DASHSCOPE_API_KEY"),
            base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"
        )
        self.model = model

        # 三层记忆
        self.short_term = ShortTermMemory(max_turns=4)
        self.summary_mem = SummaryMemory(self.client, model, buffer_size=4, summary_trigger=8)
        self.long_term = LongTermMemory(self.client, model)

    def chat(self, user_input: str) -> str:
        """一次对话的完整流程"""
        # 记录用户输入
        self.short_term.add("user", user_input)
        self.summary_mem.add("user", user_input)

        # 构建上下文：长期记忆 + 中期摘要 + 短期原始消息
        context_parts = []

        # 长期记忆注入系统层
        lt_context = self.long_term.get_context()
        if lt_context:
            context_parts.append({"role": "system", "content": f"【长期记忆】{lt_context}"})

        # 中期摘要
        context_parts.extend(self.summary_mem.get_messages())

        # 短期记忆
        context_parts.extend(self.short_term.get_messages())

        print(f"\n[Agent 内部] 发送给 LLM 的消息：")
        for msg in context_parts:
            print(f"  {msg['role']}: {msg['content'][:50]}...")

        # 调用 LLM
        response = self.client.chat.completions.create(
            model=self.model,
            messages=context_parts
        )
        answer = response.choices[0].message.content

        # 记录助手回复
        self.short_term.add("assistant", answer)
        self.summary_mem.add("assistant", answer)

        # 定期提取长期记忆（简化版：每轮都提取，生产环境应异步/定时触发）
        self.long_term.extract_and_store(self.short_term.get_messages())

        return answer

    def get_memory_state(self) -> Dict:
        """查看当前记忆状态 —— 面试调试时常用"""
        return {
            "short_term": self.short_term.get_messages(),
            "summary": self.summary_mem.summary,
            "raw_count": len(self.summary_mem.raw_messages),
            "long_term_facts": self.long_term.facts,
            "long_term_preferences": self.long_term.preferences,
        }

"""
【面试追问准备】

Q: 为什么需要三层记忆？一层不够吗？
A: 单层记忆无法兼顾"实时性"和"容量"：
   - 纯短期记忆：容量小，跨会话丢失
   - 纯长期记忆：检索有延迟，不适合当前会话上下文
   - 纯摘要：丢失细节
   三层架构（短期原始 + 中期摘要 + 长期结构化）是工程上的折中方案。

Q: 长期记忆提取的频率怎么控制？
A: 生产环境中应该异步触发：
   - 方案 A：每 N 轮对话后异步调用提取
   - 方案 B：对话结束后统一提取
   - 方案 C：实时流式提取（成本高）

Q: 多用户场景下记忆怎么隔离？
A: 每个 user_id 有独立的记忆实例；
   或者共享向量库但用 metadata 过滤（where user_id == xxx）。

Q: LangChain 的 Memory 怎么实现这些功能？
A: LangChain 提供：
   - ConversationBufferMemory（滑动窗口）
   - ConversationSummaryMemory（摘要压缩）
   - VectorStoreRetrieverMemory（向量检索）
   - EntityMemory（实体提取）
   可以组合使用实现类似效果。
"""
