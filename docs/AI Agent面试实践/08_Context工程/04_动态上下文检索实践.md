# 动态上下文检索实践（Context Retrieval）

> 只给模型"当下需要"的信息，而非全量塞入

---

## 一、为什么需要动态检索

**问题**：复杂 Agent 面临的信息爆炸
- 项目文档成千上万页
- 工具返回结果动辄数千 token
- 多轮对话历史层层累积

**直觉**：全量塞入上下文 = 让模型在 haystack 里找 needle

**正解**：按需检索，只给当前推理步骤最相关的信息

---

## 二、三种检索策略

### 策略 1：RAG 增强（已有知识 + 按需检索）

**适用场景**：Agent 需要访问外部知识库（文档、代码、数据库）

**核心流程**：
```
用户查询 → 生成检索关键词 → 向量检索/关键词检索 → Top-K 结果 → 注入上下文
```

```python
# context_retrieval_rag.py
"""
RAG 增强的动态上下文检索
结合向量检索 + 重排序，只把最相关的文档片段注入上下文
"""

import numpy as np
from typing import List, Dict
import json

class SimpleVectorStore:
    """简化版向量存储（生产环境用 Chroma/Pinecone）"""
    
    def __init__(self):
        self.documents = []  # [(text, metadata)]
        self.embeddings = []  # np.array
    
    def add(self, text: str, metadata: Dict = None, embedding=None):
        """添加文档"""
        self.documents.append((text, metadata or {}))
        # 实际生产中使用 embedding model
        # 这里简化：用 hash 模拟
        self.embeddings.append(embedding or np.random.randn(128))
    
    def search(self, query_embedding, top_k=3):
        """向量相似度搜索"""
        if not self.embeddings:
            return []
        
        # 计算余弦相似度
        scores = []
        for emb in self.embeddings:
            similarity = np.dot(query_embedding, emb) / (
                np.linalg.norm(query_embedding) * np.linalg.norm(emb)
            )
            scores.append(similarity)
        
        # 返回 Top-K
        top_indices = np.argsort(scores)[::-1][:top_k]
        return [
            {
                "text": self.documents[i][0],
                "metadata": self.documents[i][1],
                "score": float(scores[i])
            }
            for i in top_indices
        ]


class RAGContextBuilder:
    """基于 RAG 的动态上下文构建器"""
    
    def __init__(self, vector_store):
        self.vector_store = vector_store
        self.max_context_tokens = 2000  # 留给检索结果的最大 token
    
    def build(self, user_query: str, conversation_history: List[Dict] = None) -> str:
        """
        构建注入 LLM 的上下文
        1. 生成检索关键词（可用 LLM 或规则）
        2. 向量检索 Top-K 文档
        3. 组装上下文
        """
        # 步骤 1：提取检索关键词（简化版：直接用用户查询）
        # 生产环境中可以用 LLM 生成同义词、扩展词
        search_keywords = self._extract_keywords(user_query)
        
        # 步骤 2：模拟向量检索（生产环境用真实 embedding）
        # 这里简化：用关键词匹配模拟
        results = self._mock_search(search_keywords)
        
        # 步骤 3：组装上下文
        context_parts = []
        
        # 添加系统指令
        context_parts.append("## 可用参考资料（按相关性排序）")
        
        for i, result in enumerate(results, 1):
            source = result.get("metadata", {}).get("source", "未知")
            text = result["text"][:500]  # 截断控制长度
            context_parts.append(f"\n### [{i}] {source}\n{text}")
        
        context_parts.append("\n## 用户当前问题")
        context_parts.append(user_query)
        
        return "\n".join(context_parts)
    
    def _extract_keywords(self, query: str) -> List[str]:
        """提取检索关键词（简化版）"""
        # 生产环境：用 LLM 做 query understanding
        # 示例："退款政策是什么" → ["退款", "refund", "policy"]
        keywords = [w for w in query.split() if len(w) > 1]
        return keywords[:5]  # 最多 5 个关键词
    
    def _mock_search(self, keywords: List[str], top_k=3):
        """模拟检索结果"""
        # 实际生产环境调用 vector_store.search()
        all_results = self.vector_store.search(
            query_embedding=np.random.randn(128),  # 模拟
            top_k=top_k
        )
        return all_results

### 策略 2：Agentic Search（Agent 主动决定检索什么）

**适用场景**：复杂推理任务，需要多步骤信息收集

**核心思想**：不让用户查询直接决定检索，而是让 Agent 自己规划检索策略

```python
# context_retrieval_agentic.py
"""
Agentic Search：Agent 自主规划检索策略
适合：研究型任务、复杂问题解答、代码审查
"""

from typing import List, Dict
import json

class AgenticSearch:
    """
    Agent 驱动的检索流程：
    1. 分析用户问题 → 生成检索计划
    2. 执行多步检索
    3. 评估信息充分性 → 决定继续或停止
    4. 综合所有检索结果生成答案
    """
    
    def __init__(self, search_tools: List[Dict]):
        """
        search_tools: 可用的检索工具列表
        示例：[{"name": "search_docs", "description": "搜索技术文档"}]
        """
        self.tools = {t["name"]: t for t in search_tools}
        self.retrieved_info = []
    
    def plan_and_search(self, user_query: str, max_iterations=3) -> str:
        """
        主流程：规划 → 检索 → 评估 → 综合
        实际生产环境中，这里调用 LLM 做决策
        """
        print(f"🔍 用户问题: {user_query}")
        
        # 步骤 1：生成检索计划（模拟 LLM 决策）
        plan = self._generate_search_plan(user_query)
        print(f"📋 检索计划: {json.dumps(plan, ensure_ascii=False)}")
        
        # 步骤 2：执行多步检索
        for i in range(max_iterations):
            print(f"\n--- 迭代 {i+1} ---")
            
            # 决定下一步检索什么
            next_action = self._decide_next_action(plan, self.retrieved_info)
            
            if next_action["action"] == "sufficient":
                print("✅ 信息已充分，停止检索")
                break
            
            # 执行检索
            results = self._execute_search(next_action)
            self.retrieved_info.extend(results)
            print(f"📄 检索到 {len(results)} 条结果")
            
            # 评估信息质量
            assessment = self._assess_information(results, user_query)
            print(f"📊 信息评估: {assessment}")
        
        # 步骤 3：综合结果
        context = self._synthesize_context(user_query)
        return context
    
    def _generate_search_plan(self, query: str) -> List[Dict]:
        """
        生成检索计划
        实际生产环境：调用 LLM 做 query decomposition
        """
        # 简化版：按关键词拆分检索任务
        if "性能" in query or "优化" in query:
            return [
                {"query": "性能基准测试方法", "tool": "search_docs"},
                {"query": "常见性能瓶颈", "tool": "search_docs"},
                {"query": "优化案例", "tool": "search_code"}
            ]
        elif "架构" in query or "设计" in query:
            return [
                {"query": "系统架构文档", "tool": "search_docs"},
                {"query": "架构决策记录 ADR", "tool": "search_docs"}
            ]
        else:
            return [{"query": query, "tool": "search_docs"}]
    
    def _decide_next_action(self, plan: List[Dict], retrieved: List[Dict]) -> Dict:
        """
        决定下一步动作
        实际生产环境：调用 LLM 判断信息是否充分
        """
        if len(retrieved) >= 3:
            return {"action": "sufficient"}
        
        # 继续按计划检索
        remaining = [p for p in plan if not any(
            r.get("query") == p["query"] for r in retrieved
        )]
        
        if remaining:
            return {
                "action": "search",
                "query": remaining[0]["query"],
                "tool": remaining[0]["tool"]
            }
        return {"action": "sufficient"}
    
    def _execute_search(self, action: Dict) -> List[Dict]:
        """模拟执行检索"""
        query = action["query"]
        tool = action["tool"]
        
        # 模拟检索结果
        mock_results = {
            "性能基准测试方法": [
                {"text": "使用 pytest-benchmark 进行基准测试...", "source": "测试文档"},
                {"text": "关键指标：吞吐量、延迟、内存占用", "source": "性能指南"}
            ],
            "常见性能瓶颈": [
                {"text": "数据库 N+1 查询问题是常见瓶颈...", "source": "优化手册"},
                {"text": "未使用索引导致全表扫描", "source": "数据库文档"}
            ],
            "系统架构文档": [
                {"text": "系统采用微服务架构，包含 API 网关...", "source": "架构文档"}
            ]
        }
        
        return mock_results.get(query, [
            {"text": f"关于 {query} 的检索结果...", "source": tool}
        ])
    
    def _assess_information(self, results: List[Dict], query: str) -> str:
        """评估信息质量"""
        if not results:
            return "未检索到相关信息"
        
        total_length = sum(len(r["text"]) for r in results)
        if total_length < 200:
            return "信息量少，可能需要补充"
        return f"信息量充足（{total_length} 字符）"
    
    def _synthesize_context(self, user_query: str) -> str:
        """综合所有检索结果构建上下文"""
        context_parts = [
            "## 检索总结",
            f"原始问题：{user_query}",
            f"检索轮次：{len(self.retrieved_info)} 次",
            f"获取信息：{sum(len(r.get('text', '')) for r in self.retrieved_info)} 字符",
            "\n## 检索结果",
        ]
        
        for i, result in enumerate(self.retrieved_info, 1):
            source = result.get("source", "未知")
            text = result["text"][:300]
            context_parts.append(f"\n[{i}] {source}\n{text}")
        
        return "\n".join(context_parts)


# ============ 演示 ============
if __name__ == "__main__":
    # 初始化 Agentic Search
    tools = [
        {"name": "search_docs", "description": "搜索技术文档和手册"},
        {"name": "search_code", "description": "搜索代码仓库"},
        {"name": "search_logs", "description": "搜索日志和监控数据"}
    ]
    
    agent = AgenticSearch(tools)
    
    # 运行
    query = "系统最近性能下降，怎么排查？"
    context = agent.plan_and_search(query, max_iterations=3)
    
    print("\n" + "="*50)
    print("=== 最终构建的上下文 ===")
    print(context)
```

### 策略 3：分层记忆按需加载

**适用场景**：超长项目，需要维护数月甚至数年的上下文

**核心思想**：把记忆分层，按当前任务按需加载不同层级

```python
# context_retrieval_hierarchical.py
"""
分层记忆按需加载：
- L1 工作记忆：当前会话（自动加载）
- L2 任务记忆：最近相关任务（按需检索）
- L3 项目知识：项目文档和架构（按需检索）
- L4 用户偏好：编码风格、常用命令（启动时加载）
"""

import json
import os
from typing import List, Dict

class HierarchicalContextLoader:
    """分层上下文加载器"""
    
    def __init__(self, memory_dir="./.agent_memory"):
        self.memory_dir = memory_dir
        self.cache = {}  # 缓存已加载的记忆
    
    def load(self, query: str, session_history: List[Dict] = None) -> Dict:
        """
        分层加载上下文
        返回不同层级的信息，由调用者组装
        """
        context = {
            "working": [],      # L1: 工作记忆
            "episodic": [],     # L2: 情节记忆
            "semantic": [],     # L3: 语义记忆
            "procedural": []    # L4: 程序记忆
        }
        
        # L1: 工作记忆（当前会话）- 始终加载
        context["working"] = session_history or []
        
        # L2: 情节记忆（相关历史任务）- 按需检索
        context["episodic"] = self._search_episodic(query)
        
        # L3: 语义记忆（项目知识）- 按需检索
        context["semantic"] = self._search_semantic(query)
        
        # L4: 程序记忆（用户偏好）- 启动时加载，复用缓存
        context["procedural"] = self._get_procedural()
        
        return context
    
    def _search_episodic(self, query: str, top_k=2) -> List[Dict]:
        """搜索相关历史任务"""
        path = os.path.join(self.memory_dir, "episodic.json")
        if not os.path.exists(path):
            return []
        
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        # 简化版：关键词匹配
        # 生产环境：用向量检索
        results = []
        for ep in data.get("episodes", []):
            if any(kw in ep["task"].lower() for kw in query.lower().split()):
                results.append(ep)
        
        return results[:top_k]
    
    def _search_semantic(self, query: str, top_k=3) -> List[Dict]:
        """搜索项目知识"""
        path = os.path.join(self.memory_dir, "semantic.json")
        if not os.path.exists(path):
            return []
        
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        # 按类别和关键词匹配
        results = []
        for fact in data.get("facts", []):
            if any(kw in fact["content"].lower() for kw in query.lower().split()):
                results.append(fact)
        
        return results[:top_k]
    
    def _get_procedural(self) -> Dict:
        """获取用户偏好（缓存）"""
        if "procedural" in self.cache:
            return self.cache["procedural"]
        
        path = os.path.join(self.memory_dir, "procedural.json")
        if not os.path.exists(path):
            return {}
        
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        self.cache["procedural"] = data.get("preferences", {})
        return self.cache["procedural"]
    
    def assemble(self, context: Dict, max_tokens=4000) -> str:
        """
        按优先级组装上下文
        优先级：程序记忆 > 工作记忆 > 情节记忆 > 语义记忆
        """
        parts = []
        total_length = 0
        
        # 最高优先级：用户偏好（影响行为模式）
        if context["procedural"]:
            pref_text = "## 用户偏好\n" + "\n".join([
                f"- {k}: {v.get('value', '')}"
                for k, v in context["procedural"].items()
            ])
            parts.append(pref_text)
            total_length += len(pref_text)
        
        # 第二优先级：工作记忆（当前上下文）
        if context["working"]:
            work_text = "## 当前对话\n" + "\n".join([
                f"[{m.get('role')}] {str(m.get('content', ''))[:150]}"
                for m in context["working"][-5:]  # 最近 5 轮
            ])
            parts.append(work_text)
            total_length += len(work_text)
        
        # 第三优先级：情节记忆（相关经验）
        if context["episodic"]:
            for ep in context["episodic"]:
                ep_text = f"## 相关经验\n- {ep['task']} → {ep['outcome']}"
                if total_length + len(ep_text) > max_tokens:
                    break
                parts.append(ep_text)
                total_length += len(ep_text)
        
        # 最低优先级：语义记忆（项目知识）
        if context["semantic"]:
            for fact in context["semantic"]:
                fact_text = f"## 项目知识\n- {fact['content']}"
                if total_length + len(fact_text) > max_tokens:
                    break
                parts.append(fact_text)
                total_length += len(fact_text)
        
        return "\n\n".join(parts)

**面试要点**：
- 分层加载 = 按优先级分配注意力预算
- 用户偏好（程序记忆）影响最大，优先加载
- 工作记忆保证连贯性，情节记忆提供经验参考

## 三、三种策略对比

| 策略 | 适用场景 | 实现复杂度 | 检索精准度 | 延迟 |
|------|---------|-----------|-----------|------|
| **RAG 增强** | 知识库问答、文档查询 | 中 | 中 | 低 |
| **Agentic Search** | 研究型任务、复杂排查 | 高 | 高 | 中 |
| **分层记忆** | 长期项目、跨会话任务 | 高 | 高 | 低 |

## 四、面试答题模板

**Q：Agent 如何避免上下文信息过载？**

**A**：
> 我采用"按需检索"策略。不把所有知识全量塞入上下文，而是让 Agent 自主决定需要什么信息。具体有三层：RAG 增强解决外部知识检索；Agentic Search 让 Agent 规划多步检索；分层记忆按需加载不同层级的信息。核心思想是"只给模型当下需要的信息"，把有限的上下文窗口留给最高信号密度的内容。

**Q：RAG 和 Context Retrieval 有什么区别？**

**A**：
> RAG 是 Context Retrieval 的一种实现方式。RAG 侧重"从外部文档检索相关知识"；Context Retrieval 是更广的概念，包括从对话历史、任务记录、用户偏好等多源动态获取信息。在实际系统中，RAG 提供事实知识，分层记忆提供任务连续性，Agentic Search 提供推理深度，三者互补。
