    结构化笔记与持久化记忆（Structured Note-taking）¶

    让 Agent 拥有"跨会话记忆"的核心技术

    一、为什么需要持久化记忆¶

    Compaction 解决了"上下文窗口内"的问题，但无法跨越会话边界： - Agent 重启后，历史对话全部丢失 - 多轮任务中，早期关键信息被压缩后可能失真 - 复杂项目需要跨天、跨周持续跟踪

    Structured Note-taking = Agent 主动写笔记到文件系统，后续读取恢复上下文

    二、核心设计原则¶

    原则 1：笔记由 Agent 自己写，而非人类预设¶

    反模式：人类写死记忆模板，Agent 被动填充。

    正模式：Agent 根据任务进展，自主决定记录什么、如何组织。

    # 反模式 ❌：人类预设模板
MEMORY_TEMPLATE = """
## 用户信息
- 姓名：{name}
- 偏好：{preference}
"""

# 正模式 ✅：Agent 自主记录
agent_note = """
用户正在重构数据处理模块。
关键决策：使用 Pandas chunk 处理大文件。
待办：1) 异常值检测 2) 性能基准测试
注意事项：当前内存占用仍偏高，考虑流式处理
"""

    原则 2：笔记要结构化，便于快速恢复¶

    推荐格式：Markdown + 特定标记，兼顾人类可读和机器解析。

    原则 3：分层存储，按需加载¶

    层级 存储位置 内容 生命周期
    会话级 内存 / 上下文 当前对话历史 单次会话
    任务级 notes.md / memory.json 任务目标、关键决策、待办 跨会话
    项目级 .agent/memory/ 项目架构、常见问题、API 文档 长期
    用户级 ~/.agent_config/ 用户偏好、常用命令、编码风格 永久

    三、Python 代码实践¶

    基础版：TODO 列表模式¶

    # memory_todo.py
"""
基础持久化记忆：TODO 列表跟踪任务进展
适合：单任务、明确目标、步骤清晰的场景
"""

import json
import os
from datetime import datetime

class TodoMemory:
    """TODO 式记忆管理"""

    def __init__(self, task_name, storage_dir="./agent_memory"):
        self.task_name = task_name
        self.storage_path = os.path.join(storage_dir, f"{task_name}_todo.json")
        os.makedirs(storage_dir, exist_ok=True)
        self.todos = self._load()

    def _load(self):
        """从磁盘加载 TODO 列表"""
        if os.path.exists(self.storage_path):
            with open(self.storage_path, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {
            "created_at": datetime.now().isoformat(),
            "task": self.task_name,
            "items": [],
            "completed": [],
            "notes": []
        }

    def _save(self):
        """保存到磁盘"""
        with open(self.storage_path, 'w', encoding='utf-8') as f:
            json.dump(self.todos, f, ensure_ascii=False, indent=2)

    def add(self, item, priority="medium"):
        """添加待办事项"""
        self.todos["items"].append({
            "content": item,
            "priority": priority,
            "added_at": datetime.now().isoformat(),
            "done": False
        })
        self._save()

    def complete(self, item_index):
        """标记完成"""
        if 0 <= item_index < len(self.todos["items"]):
            item = self.todos["items"].pop(item_index)
            item["completed_at"] = datetime.now().isoformat()
            self.todos["completed"].append(item)
            self._save()

    def add_note(self, note):
        """添加过程笔记"""
        self.todos["notes"].append({
            "content": note,
            "time": datetime.now().isoformat()
        })
        self._save()

    def get_summary(self):
        """获取当前状态摘要（用于注入上下文）"""
        pending = [i["content"] for i in self.todos["items"] if not i["done"]]
        notes = [n["content"] for n in self.todos["notes"][-5:]]  # 最近 5 条

        summary = f"""## 任务：{self.task_name}
- **待办** ({len(pending)}项): {', '.join(pending) if pending else '无'}
- **已完成** ({len(self.todos['completed'])}项)
- **最近笔记**: {'; '.join(notes) if notes else '无'}
- **创建于**: {self.todos['created_at']}
"""
        return summary

    def clear(self):
        """任务完成后清理"""
        if os.path.exists(self.storage_path):
            os.remove(self.storage_path)


# ============ 演示 ============
if __name__ == "__main__":
    memory = TodoMemory("重构数据处理模块")

    # Agent 运行过程中记录
    memory.add("分析当前 data_processor.py 的实现", "high")
    memory.add("设计 chunk 处理方案", "high")
    memory.add("编写单元测试", "medium")
    memory.add_note("发现当前使用 pd.read_csv() 一次性加载 1GB 文件")
    memory.add_note("内存占用峰值 8GB，需要优化")

    # 完成一项
    memory.complete(0)
    memory.add_note("已完成分析，确认使用 chunk_size=10000")

    # 获取摘要（用于注入上下文）
    print("=== 当前状态摘要 ===")
    print(memory.get_summary())

    # 模拟会话重启后恢复
    print("\n=== 模拟重启后加载 ===")
    memory2 = TodoMemory("重构数据处理模块")
    print(memory2.get_summary())

    # 清理
    memory.clear()

    面试要点： - 最简单的持久化记忆实现 - 适合：任务明确、步骤清晰的场景 - 不足：无法记录复杂关系、非结构化信息

    进阶版：Markdown 笔记模式¶

    # memory_markdown.py
"""
Markdown 笔记模式：Agent 自主维护项目笔记
适合：复杂项目、需要跨天持续跟踪的场景
"""

import os
import re
from datetime import datetime

MARKDOWN_TEMPLATE = """# {project_name} - Agent 笔记

> 本文件由 AI Agent 自动维护，记录项目关键信息
> 最后更新：{last_updated}

## 项目目标
{goals}

## 关键决策
{decisions}

## 当前状态
{status}

## 待办事项
{todos}

## 已知问题
{issues}

## 技术笔记
{notes}

## 文件清单
{files}

## 会话历史
{sessions}
"""


class MarkdownMemory:
    """Markdown 格式记忆管理"""

    def __init__(self, project_name, memory_dir="./.agent_memory"):
        self.project_name = project_name
        self.memory_dir = memory_dir
        self.file_path = os.path.join(memory_dir, f"{project_name}_notes.md")
        os.makedirs(memory_dir, exist_ok=True)

        if not os.path.exists(self.file_path):
            self._create_empty()

    def _create_empty(self):
        """创建空笔记文件"""
        content = MARKDOWN_TEMPLATE.format(
            project_name=self.project_name,
            last_updated=datetime.now().isoformat(),
            goals="*待补充*",
            decisions="- 暂无",
            status="*进行中*",
            todos="- [ ] 待办事项",
            issues="- 暂无",
            notes="- 暂无",
            files="- 暂无",
            sessions=f"- {datetime.now().strftime('%Y-%m-%d')}: 项目启动"
        )
        with open(self.file_path, 'w', encoding='utf-8') as f:
            f.write(content)

    def _read(self):
        """读取笔记内容"""
        with open(self.file_path, 'r', encoding='utf-8') as f:
            return f.read()

    def _write(self, content):
        """写入笔记"""
        with open(self.file_path, 'w', encoding='utf-8') as f:
            f.write(content)

    def _update_section(self, section, new_content):
        """更新特定章节"""
        content = self._read()
        pattern = f"(## {section}\\n)(.*?)(?=\\n## |\\Z)"
        replacement = f"\\1{new_content}\\n"
        updated = re.sub(pattern, replacement, content, flags=re.DOTALL)
        self._write(updated)

    def add_decision(self, decision, reason=""):
        """记录关键决策"""
        entry = f"- **{datetime.now().strftime('%m-%d')}**: {decision}"
        if reason:
            entry += f" （原因：{reason}）"
        entry += "\n"

        content = self._read()
        # 在"## 关键决策"后追加
        updated = re.sub(
            r"(## 关键决策\n)(.*?)(?=\n## )",
            lambda m: f"{m.group(1)}{m.group(2)}{entry}",
            content,
            flags=re.DOTALL
        )
        self._write(updated)

    def add_todo(self, todo, checked=False):
        """添加待办"""
        checkbox = "[x]" if checked else "[ ]"
        entry = f"- {checkbox} {todo}\n"
        self._update_section("待办事项", self._read_section("待办事项") + entry)

    def complete_todo(self, todo_text):
        """标记待办完成"""
        content = self._read()
        updated = content.replace(
            f"- [ ] {todo_text}",
            f"- [x] {todo_text} ✓{datetime.now().strftime('%m-%d')}"
        )
        self._write(updated)

    def add_note(self, note):
        """添加技术笔记"""
        entry = f"- **{datetime.now().strftime('%m-%d %H:%M')}**: {note}\n"
        self._update_section("技术笔记", self._read_section("技术笔记") + entry)

    def _read_section(self, section):
        """读取特定章节内容"""
        content = self._read()
        match = re.search(f"## {section}\\n(.*?)(?=\\n## |\\Z)", content, re.DOTALL)
        return match.group(1) if match else ""

    def get_context_summary(self, max_chars=2000):
        """获取用于注入上下文的摘要"""
        content = self._read()
        # 提取关键部分：目标、关键决策、当前状态、待办
        sections = ["项目目标", "关键决策", "当前状态", "待办事项", "已知问题"]
        summary = []
        for section in sections:
            section_content = self._read_section(section).strip()
            if section_content and section_content != "*待补充*":
                summary.append(f"## {section}\n{section_content}")

        result = "\n\n".join(summary)
        return result[:max_chars] if len(result) > max_chars else result

    def add_session(self, summary):
        """记录会话摘要"""
        entry = f"- {datetime.now().strftime('%Y-%m-%d %H:%M')}: {summary}\n"
        self._update_section("会话历史", self._read_section("会话历史") + entry)


# ============ 演示 ============
if __name__ == "__main__":
    memory = MarkdownMemory("数据处理模块重构")

    # 记录关键决策
    memory.add_decision("使用 Pandas chunk 处理大文件", "内存占用从 8GB 降至 1GB")
    memory.add_decision("chunk_size = 10000", "平衡内存和速度的最佳点")

    # 记录待办
    memory.add_todo("实现异常值检测逻辑")
    memory.add_todo("编写性能基准测试")
    memory.add_todo("更新文档")

    # 记录笔记
    memory.add_note("发现 pd.read_csv 的 dtype 推断在 chunk 模式下不一致")
    memory.add_note("考虑使用 pyarrow 引擎替代默认引擎")

    # 完成一项
    memory.complete_todo("实现异常值检测逻辑")

    # 获取上下文摘要（注入 LLM 上下文）
    print("=== 上下文摘要 ===")
    print(memory.get_context_summary())

    # 查看完整笔记
    print("\n=== 完整笔记文件 ===")
    print(memory._read())

    # 清理
    import shutil
    shutil.rmtree("./.agent_memory")

    面试要点： - Agent 自主维护 Markdown，人类可随时查看 - 结构化但灵活，适合复杂项目 - 关键：get_context_summary() 按需提取，控制注入上下文的 token 数

    生产级：分层记忆系统¶

    # memory_hierarchical.py
"""
分层记忆系统：会话级 + 任务级 + 项目级 + 用户级
适合：生产级 Agent，需要长期运行的复杂系统
"""

import json
import os
from datetime import datetime, timedelta

class HierarchicalMemory:
    """
    四层记忆架构：
    1. Working Memory（工作记忆）- 当前会话
    2. Episodic Memory（情节记忆）- 任务历史
    3. Semantic Memory（语义记忆）- 项目知识
    4. Procedural Memory（程序记忆）- 用户偏好/常用模式
    """

    def __init__(self, project_name, base_dir="./.agent_memory"):
        self.project_name = project_name
        self.base_dir = base_dir

        # 四层存储路径
        self.working_path = os.path.join(base_dir, "working.json")      # 会话级
        self.episodic_path = os.path.join(base_dir, "episodic.json")   # 任务级
        self.semantic_path = os.path.join(base_dir, "semantic.json")   # 项目级
        self.procedural_path = os.path.join(base_dir, "procedural.json") # 用户级

        os.makedirs(base_dir, exist_ok=True)
        self._init_storage()

    def _init_storage(self):
        """初始化存储文件"""
        defaults = {
            self.working_path: {"session_id": "", "messages": [], "created_at": ""},
            self.episodic_path: {"episodes": []},
            self.semantic_path: {"facts": [], "concepts": {}},
            self.procedural_path: {"preferences": {}, "patterns": []}
        }
        for path, default in defaults.items():
            if not os.path.exists(path):
                with open(path, 'w', encoding='utf-8') as f:
                    json.dump(default, f, ensure_ascii=False, indent=2)

    # ========== Working Memory（会话级）==========
    def working_add(self, message):
        """添加当前会话消息"""
        data = self._load(self.working_path)
        data["messages"].append({
            **message,
            "timestamp": datetime.now().isoformat()
        })
        self._save(self.working_path, data)

    def working_get_recent(self, n=10):
        """获取最近 N 条"""
        data = self._load(self.working_path)
        return data["messages"][-n:]

    # ========== Episodic Memory（任务级）==========
    def episodic_save(self, task_summary, outcome, lessons=""):
        """保存任务 episode"""
        data = self._load(self.episodic_path)
        data["episodes"].append({
            "task": task_summary,
            "outcome": outcome,
            "lessons": lessons,
            "timestamp": datetime.now().isoformat()
        })
        # 只保留最近 50 个 episode
        data["episodes"] = data["episodes"][-50:]
        self._save(self.episodic_path, data)

    def episodic_search(self, keyword, n=3):
        """搜索相关 episode"""
        data = self._load(self.episodic_path)
        matches = [e for e in data["episodes"] if keyword.lower() in e["task"].lower()]
        return sorted(matches, key=lambda x: x["timestamp"], reverse=True)[:n]

    # ========== Semantic Memory（项目级）==========
    def semantic_learn(self, fact, category="general"):
        """学习项目知识"""
        data = self._load(self.semantic_path)
        data["facts"].append({
            "content": fact,
            "category": category,
            "learned_at": datetime.now().isoformat(),
            "access_count": 0
        })
        self._save(self.semantic_path, data)

    def semantic_query(self, category=None, n=5):
        """查询知识（可按类别筛选）"""
        data = self._load(self.semantic_path)
        facts = data["facts"]
        if category:
            facts = [f for f in facts if f["category"] == category]
        # 按访问频次排序（常用知识优先）
        return sorted(facts, key=lambda x: x.get("access_count", 0), reverse=True)[:n]

    # ========== Procedural Memory（用户级）==========
    def procedural_learn_preference(self, key, value):
        """学习用户偏好"""
        data = self._load(self.procedural_path)
        data["preferences"][key] = {
            "value": value,
            "learned_at": datetime.now().isoformat(),
            "confidence": 1
        }
        self._save(self.procedural_path, data)

    def procedural_get_preference(self, key, default=None):
        """获取用户偏好"""
        data = self._load(self.procedural_path)
        pref = data["preferences"].get(key)
        return pref["value"] if pref else default

    # ========== 工具方法 ==========
    def _load(self, path):
        with open(path, 'r', encoding='utf-8') as f:
            return json.load(f)

    def _save(self, path, data):
        with open(path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def build_context(self, query="", max_tokens=4000):
        """
        构建注入 LLM 的上下文
        智能选择：工作记忆 + 相关情节 + 相关知识
        """
        parts = []

        # 1. 工作记忆（最近 5 轮）
        working = self.working_get_recent(5)
        parts.append("## 当前会话\n" + "\n".join([
            f"[{m.get('role')}] {str(m.get('content', ''))[:200]}"
            for m in working
        ]))

        # 2. 相关情节（基于查询关键词）
        if query:
            episodes = self.episodic_search(query, n=2)
            if episodes:
                parts.append("## 相关经验\n" + "\n".join([
                    f"- {e['task']} → {e['outcome']}"
                    for e in episodes
                ]))

        # 3. 高频知识
        facts = self.semantic_query(n=3)
        if facts:
            parts.append("## 项目知识\n" + "\n".join([
                f"- {f['content']}"
                for f in facts
            ]))

        # 4. 用户偏好
        prefs = self._load(self.procedural_path)["preferences"]
        if prefs:
            parts.append("## 用户偏好\n" + "\n".join([
                f"- {k}: {v['value']}"
                for k, v in list(prefs.items())[:3]
            ]))

        return "\n\n".join(parts)

    四、面试答题模板¶

    Q：Agent 如何做到"跨会话记忆"？

    A：

    我采用分层记忆架构。工作记忆保存当前会话的最近消息；情节记忆记录任务历史和教训；语义记忆存储项目知识；程序记忆学习用户偏好。每次会话开始时，根据用户查询从长期记忆中检索相关信息，注入到当前上下文中。Claude Plays Pokémon 的案例证明，Agent 通过维护笔记能在数千步后保持策略连贯性。

    Q：持久化记忆和 RAG 有什么区别？

    A：

    RAG 解决的是"外部知识注入"问题，把文档、数据库中的信息拉入上下文；持久化记忆解决的是"跨会话状态保持"问题，让 Agent 记住之前的对话、决策和偏好。两者互补：RAG 提供事实知识，持久化记忆提供任务连续性。 ```
